\documentclass[preview]{standalone}

\usepackage{gitdags}
\begin{document}
\begin{figure}
  \begin{subfigure}[b]{\textwidth}
    \centering
    \begin{tikzpicture}
      % Commit DAG
      \gitDAG[grow right sep = 2em]{
        A -- B -- C -- { 
          D -- E,
          F,
        }
      };
      % Branch
      \gitbranch
        {experimental}     % node name and text 
        {below=of F} % node placement
        {F}          % target
      \gitbranch
        {master}     % node name and text 
        {above=of E} % node placement
        {E}          % target
      % HEAD reference
      \gitHEAD
        {above=of master} % node placement
        {master}          % target
    \end{tikzpicture}
  \end{subfigure}
\end{figure}
